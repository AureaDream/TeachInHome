// 云函数入口文件
const cloud = require('wx-server-sdk')
// 引入腾讯云短信 SDK
const tencentcloud = require('tencentcloud-sdk-nodejs')
const SmsClient = tencentcloud.sms.v20190711.Client

cloud.init({
  env: 'cloud1-3gqdqvkpbeab224c'
})

// 云函数入口函数
exports.main = async (event, context) => {
  const wxContext = cloud.getWXContext()
  const db = cloud.database()
  
  // 获取手机号
  const { phone } = event
  
  if (!phone || !/^1[3-9]\d{9}$/.test(phone)) {
    return {
      success: false,
      message: '手机号格式不正确'
    }
  }
  
  try {
    // 生成6位随机验证码
    const code = Math.floor(100000 + Math.random() * 900000).toString()

    // 读取腾讯云短信配置（优先使用环境变量）
    const secretId = process.env.TENCENTCLOUD_SECRET_ID
    const secretKey = process.env.TENCENTCLOUD_SECRET_KEY
    const smsSdkAppid = process.env.SMS_SDK_APPID
    const templateId = process.env.SMS_TEMPLATE_ID
    const smsSign = process.env.SMS_SIGN // 国内短信为必填

    if (!secretId || !secretKey || !smsSdkAppid || !templateId || !smsSign) {
      console.error('短信配置缺失，请在云函数环境变量中设置 TENCENTCLOUD_SECRET_ID、TENCENTCLOUD_SECRET_KEY、SMS_SDK_APPID、SMS_TEMPLATE_ID、SMS_SIGN')
      return {
        success: false,
        message: '短信配置缺失，请联系管理员配置短信参数'
      }
    }

    // 构造 E.164 手机号（国内默认 +86）
    const phoneE164 = phone.startsWith('+') ? phone : `+86${phone}`

    // 初始化短信客户端
    const client = new SmsClient({
      credential: { secretId, secretKey },
      region: '', // 短信无需填写 region
      profile: { httpProfile: { endpoint: 'sms.tencentcloudapi.com' } }
    })

    // 发送短信
    const sessionCtx = `outsid_${Date.now()}_${Math.floor(Math.random() * 10000)}`
    const params = {
      SmsSdkAppid: smsSdkAppid,
      Sign: smsSign,
      TemplateID: templateId,
      PhoneNumberSet: [phoneE164],
      TemplateParamSet: [code],
      SessionContext: sessionCtx
    }

    const smsResult = await client.SendSms(params)
    const status = smsResult && smsResult.SendStatusSet && smsResult.SendStatusSet[0]

    if (!status || status.Code !== 'Ok') {
      console.error('短信发送失败:', JSON.stringify(smsResult, null, 2))
      return {
        success: false,
        message: `短信发送失败：${status ? status.Message : '未知错误'}`
      }
    }

    // 短信发送成功后，保存验证码到数据库（10分钟有效）
    const codeCollection = db.collection('sms_codes')
    const existingCode = await codeCollection.where({ phone: phone }).get()
    const now = new Date()
    const expireTime = new Date(now.getTime() + 10 * 60 * 1000)

    if (existingCode.data.length > 0) {
      await codeCollection.doc(existingCode.data[0]._id).update({
        data: { code, createTime: now, expireTime, used: false }
      })
    } else {
      await codeCollection.add({
        data: { phone, code, createTime: now, expireTime, used: false }
      })
    }

    return {
      success: true,
      message: '验证码发送成功'
    }
  } catch (err) {
    console.error('发送验证码失败', err)
    return {
      success: false,
      message: '发送验证码失败，请稍后重试',
      error: err
    }
  }
}
